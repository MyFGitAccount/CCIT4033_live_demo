-- Add sample courses
INSERT INTO Course (CourseCode, CourseName, Credits, DepartmentID, IsActive, IsDeleted) VALUES
('CS101', 'Introduction to Programming', 3, 1, 1, 0),
('CS201', 'Data Structures', 3, 1, 1, 0),
('CS301', 'Database Systems', 3, 1, 1, 0);

-- Add prerequisite: CS201 requires CS101
INSERT INTO CoursePrerequisite (CourseID, PrerequisiteCourseID, MinGradeRequired)
SELECT c1.CourseID, c2.CourseID, 60
FROM Course c1, Course c2
WHERE c1.CourseCode = 'CS201' AND c2.CourseCode = 'CS101';

-- Add prerequisite: CS301 requires CS201
INSERT INTO CoursePrerequisite (CourseID, PrerequisiteCourseID, MinGradeRequired)
SELECT c1.CourseID, c2.CourseID, 60
FROM Course c1, Course c2
WHERE c1.CourseCode = 'CS301' AND c2.CourseCode = 'CS201';

-- Create a section for CS201
INSERT INTO Section (SectionCode, SectionName, TermID, CourseID, MaxCapacity, Status)
SELECT 'CS201A', 'Section A', TermID, CourseID, 30, 'Active'
FROM Course, AcademicTerm
WHERE Course.CourseCode = 'CS201' AND AcademicTerm.IsCurrent = 1;
