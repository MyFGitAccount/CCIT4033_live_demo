-- =====================================================
-- ENHANCED AUDIT LOG (Tamper-proof)
-- =====================================================
CREATE TABLE IF NOT EXISTS AuditLog (
    LogID INTEGER PRIMARY KEY AUTOINCREMENT,
    TableName TEXT NOT NULL,
    RecordID INTEGER NOT NULL,
    Action TEXT CHECK(Action IN ('INSERT', 'UPDATE', 'DELETE', 'SOFT_DELETE', 'WARNING', 'GRADE_RECALC', 'WAITLIST_PROMOTION')),
    OldValue TEXT,
    NewValue TEXT,
    TriggerName TEXT NOT NULL,
    ChangedBy TEXT DEFAULT 'SYSTEM',
    ChangedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    IPAddress TEXT,
    SessionID TEXT,
    Checksum TEXT
);
CREATE TRIGGER prevent_auditlog_modification BEFORE UPDATE ON AuditLog
BEGIN SELECT RAISE(ABORT, 'AuditLog records cannot be modified - they are immutable'); END;
CREATE TRIGGER prevent_auditlog_deletion BEFORE DELETE ON AuditLog
BEGIN SELECT RAISE(ABORT, 'AuditLog records cannot be deleted - they are immutable'); END;

CREATE VIEW GraduationAudit AS
SELECT
    s.StudentID,
    s.StudentNumber,
    s.FirstName || ' ' || s.LastName AS StudentName,
    s.TotalCreditsEarned,
    s.GraduationThreshold,
    s.CumulativeGPA,
    s.SemesterGPA,
    s.HasOutstandingFees,
    s.AcademicStanding,
    (s.GraduationThreshold - s.TotalCreditsEarned) AS CreditsRemaining,
    CASE
        WHEN s.TotalCreditsEarned >= s.GraduationThreshold
             AND s.CumulativeGPA >= (SELECT CAST(SettingValue AS REAL) FROM SystemSettings WHERE SettingKey = 'GRADUATION_MIN_GPA')
             AND s.HasOutstandingFees = 0
             AND s.AcademicStanding NOT IN ('Academic Probation', 'Critical')
        THEN 'Eligible for Graduation'
        WHEN s.TotalCreditsEarned >= s.GraduationThreshold THEN 'Meets Credit Requirement - Check Other Conditions'
        WHEN s.CumulativeGPA < (SELECT CAST(SettingValue AS REAL) FROM SystemSettings WHERE SettingKey = 'GRADUATION_MIN_GPA')
        THEN 'GPA Requirement Not Met'
        WHEN s.HasOutstandingFees = 1 THEN 'Outstanding Fees'
        ELSE 'Not Eligible'
    END AS GraduationEligibility,
    CASE
        WHEN s.CumulativeGPA < (SELECT CAST(SettingValue AS REAL) FROM SystemSettings WHERE SettingKey = 'GRADUATION_MIN_GPA')
        THEN ROUND((SELECT CAST(SettingValue AS REAL) FROM SystemSettings WHERE SettingKey = 'GRADUATION_MIN_GPA') - s.CumulativeGPA, 2)
        ELSE 0
    END AS GPAShortfall,
    (SELECT COUNT(*) FROM CourseRequired cr
     WHERE cr.DepartmentID = s.DepartmentID
     AND NOT EXISTS (
         SELECT 1 FROM Enrollment e
         JOIN Section sec ON e.SectionID = sec.SectionID
         WHERE e.StudentID = s.StudentID
           AND e.Status = 'Completed'
           AND sec.CourseID = cr.CourseID
     )) AS MissingRequiredCourses
FROM Student s
WHERE s.Status IN ('Active', 'Academic Probation', 'Academic Warning')
  AND s.IsDeleted = 0;

CREATE VIEW StudentRiskAssessment AS
SELECT
    s.StudentID,
    s.StudentNumber,
    s.FirstName || ' ' || s.LastName AS StudentName,
    s.CumulativeGPA,
    s.SemesterGPA,
    s.AttendancePercentage,
    s.AcademicStanding,
    s.PerformanceFlag,
    s.HasOutstandingFees,
    CASE
        WHEN s.TotalCreditsEarned = 0 THEN 'Low Risk'
        WHEN s.CumulativeGPA < 2.0 AND s.AttendancePercentage < 75 THEN 'Critical Risk'
        WHEN s.CumulativeGPA < 2.0 OR s.AttendancePercentage < 75 THEN 'High Risk'
        WHEN s.CumulativeGPA < 2.5 OR s.HasOutstandingFees = 1 OR s.PerformanceFlag = 'At Risk' THEN 'Moderate Risk'
        ELSE 'Low Risk'
    END AS RiskLevel,
    CASE
        WHEN s.TotalCreditsEarned = 0 THEN 'New student – no assessment yet'
        WHEN s.CumulativeGPA < 2.0 AND s.AttendancePercentage < 75
        THEN 'Immediate academic intervention and attendance counseling required'
        WHEN s.CumulativeGPA < 2.0
        THEN 'Academic probation - mandatory advisor meeting'
        WHEN s.AttendancePercentage < 75
        THEN 'Attendance warning - required to improve attendance'
        WHEN s.HasOutstandingFees = 1
        THEN 'Financial hold - clear fees before next registration'
        ELSE 'Continue monitoring'
    END AS Recommendation,
    (SELECT COUNT(*) FROM Enrollment e WHERE e.StudentID = s.StudentID AND e.Status = 'Withdrawn') AS WithdrawnCoursesCount,
    (SELECT COUNT(*) FROM Notification n
     WHERE n.UserID = (SELECT UserID FROM SystemUser WHERE Username = 'STU_' || s.StudentNumber)
     AND n.IsRead = 0 AND n.Priority IN ('High', 'Urgent')) AS PendingAlerts
FROM Student s
WHERE s.Status IN ('Active', 'Academic Probation', 'Academic Warning')
  AND s.IsDeleted = 0
ORDER BY
    CASE RiskLevel
        WHEN 'Critical Risk' THEN 1
        WHEN 'High Risk' THEN 2
        WHEN 'Moderate Risk' THEN 3
        ELSE 4
    END;

-- Trigger: Ensure at least one current term exists & enforce single current term
CREATE TRIGGER academic_term_after_insert
AFTER INSERT ON AcademicTerm
BEGIN
    UPDATE AcademicTerm
    SET IsCurrent = 1
    WHERE TermID = NEW.TermID
    AND DATE('now') BETWEEN StartDate AND EndDate
    AND (SELECT COUNT(*) FROM AcademicTerm WHERE IsCurrent = 1) = 0;

    UPDATE AcademicTerm
    SET IsCurrent = 1
    WHERE TermID = (SELECT TermID FROM AcademicTerm WHERE StartDate <= DATE('now') ORDER BY StartDate DESC LIMIT 1)
    AND (SELECT COUNT(*) FROM AcademicTerm WHERE IsCurrent = 1) = 0;
END;

-- Trigger: Enforce single current term on updates
CREATE TRIGGER academic_term_before_update
BEFORE UPDATE OF IsCurrent ON AcademicTerm
WHEN NEW.IsCurrent = 1 AND OLD.IsCurrent = 0
BEGIN
    UPDATE AcademicTerm SET IsCurrent = 0 WHERE IsCurrent = 1 AND TermID != NEW.TermID;
END;

-- Trigger: Auto-archive old terms
CREATE TRIGGER academic_term_auto_archive
AFTER UPDATE ON AcademicTerm
BEGIN
    UPDATE AcademicTerm
    SET IsArchived = 1, IsActive = 0
    WHERE EndDate < DATE('now', '-90 days') AND IsArchived = 0;
END;

-- Trigger: Enforce budget constraint
CREATE TRIGGER dept_budget_hard_stop
BEFORE UPDATE OF BudgetUsed ON Department
WHEN NEW.BudgetUsed > NEW.Budget
BEGIN
    SELECT RAISE(ABORT, 'Budget would be exceeded. Current Budget: ' || NEW.Budget || ', Attempted Used: ' || NEW.BudgetUsed);
END;

-- Trigger: Cascading soft delete for departments
CREATE TRIGGER dept_cascading_soft_delete
BEFORE UPDATE OF IsDeleted ON Department
WHEN NEW.IsDeleted = 1 AND OLD.IsDeleted = 0
BEGIN
    UPDATE Course SET IsDeleted = 1, IsActive = 0, DeletedAt = CURRENT_TIMESTAMP, DeletedBy = NEW.DeletedBy
    WHERE DepartmentID = OLD.DeptID AND IsDeleted = 0;

    UPDATE Student SET IsDeleted = 1, Status = 'Archived', DeletedAt = CURRENT_TIMESTAMP, DeletedBy = NEW.DeletedBy
    WHERE DepartmentID = OLD.DeptID AND IsDeleted = 0 AND Status IN ('Active', 'Academic Probation', 'Academic Warning');

    UPDATE Department SET HeadTeacherID = NULL WHERE DeptID = OLD.DeptID;

    INSERT INTO AuditLog (TableName, RecordID, Action, OldValue, NewValue, TriggerName, ChangedBy)
    VALUES ('Department', OLD.DeptID, 'SOFT_DELETE', 'Active', 'Deleted',
            'dept_cascading_soft_delete', NEW.DeletedBy);
END;

-- Trigger: Update workload after section insert
CREATE TRIGGER teacher_after_section_insert
AFTER INSERT ON Section
WHEN (SELECT TriggersEnabled FROM TriggerControl WHERE ControlID = 1) = 1 AND NEW.TeacherID IS NOT NULL
BEGIN
    UPDATE Teacher
    SET WeeklyWorkloadHours = (
        SELECT COALESCE(SUM(
            CASE
                WHEN s.Schedule LIKE '%Monday%' THEN 3
                WHEN s.Schedule LIKE '%Tuesday%' THEN 3
                WHEN s.Schedule LIKE '%Wednesday%' THEN 3
                WHEN s.Schedule LIKE '%Thursday%' THEN 3
                WHEN s.Schedule LIKE '%Friday%' THEN 3
                ELSE 2
            END
        ), 0)
        FROM Section s
        WHERE s.TeacherID = Teacher.TeacherID AND s.Status = 'Active'
    ),
    LastModifiedDate = CURRENT_TIMESTAMP
    WHERE TeacherID = NEW.TeacherID;

    SELECT CASE
        WHEN (SELECT WeeklyWorkloadHours FROM Teacher WHERE TeacherID = NEW.TeacherID) >
             (SELECT CAST(SettingValue AS INTEGER) FROM SystemSettings WHERE SettingKey = 'MAX_TEACHER_WORKLOAD_HOURS')
        THEN RAISE(ABORT, 'Teacher weekly workload would exceed maximum allowed hours')
    END;

    SELECT CASE
        WHEN (SELECT COUNT(*) FROM Section
              WHERE TeacherID = NEW.TeacherID AND TermID = NEW.TermID AND Status != 'Completed') >
             (SELECT CAST(SettingValue AS INTEGER) FROM SystemSettings WHERE SettingKey = 'MAX_SECTIONS_PER_TEACHER')
        THEN RAISE(ABORT, 'Teacher cannot teach more than ' ||
            (SELECT SettingValue FROM SystemSettings WHERE SettingKey = 'MAX_SECTIONS_PER_TEACHER') ||
            ' active sections per semester')
    END;

    SELECT CASE
        WHEN EXISTS (
            SELECT 1 FROM Section s
            JOIN Course c1 ON s.CourseID = c1.CourseID
            JOIN Course c2 ON c2.CourseID = NEW.CourseID
            WHERE s.TeacherID = NEW.TeacherID
              AND s.TermID = NEW.TermID
              AND s.Status != 'Completed'
              AND c1.DepartmentID != c2.DepartmentID
        ) THEN RAISE(ABORT, 'Teacher cannot teach sections in different departments in the same semester')
    END;
END;

-- Trigger: Update student performance metrics
CREATE TRIGGER student_performance_update
AFTER UPDATE OF CumulativeGPA, AttendancePercentage, SemesterGPA ON Student
WHEN (SELECT TriggersEnabled FROM TriggerControl WHERE ControlID = 1) = 1
BEGIN
    UPDATE Student
    SET
        AcademicStanding = CASE
            WHEN NEW.CumulativeGPA >= 3.8 THEN 'Dean''s List'
            WHEN NEW.CumulativeGPA >= 3.5 THEN 'Honors'
            WHEN NEW.CumulativeGPA >= (SELECT CAST(SettingValue AS REAL) FROM SystemSettings WHERE SettingKey = 'GPA_GOOD_STANDING') THEN 'Good Standing'
            WHEN NEW.CumulativeGPA >= (SELECT CAST(SettingValue AS REAL) FROM SystemSettings WHERE SettingKey = 'GPA_PROBATION_THRESHOLD') THEN 'Academic Warning'
            ELSE 'Academic Probation'
        END,
        Status = CASE
            WHEN NEW.AcademicStanding = 'Academic Probation' AND NEW.Status = 'Active' THEN 'Academic Probation'
            WHEN NEW.AcademicStanding = 'Good Standing' AND NEW.Status IN ('Academic Probation', 'Academic Warning') THEN 'Active'
            ELSE NEW.Status
        END,
        PerformanceFlag = CASE
            WHEN NEW.CumulativeGPA >= (SELECT CAST(SettingValue AS REAL) FROM SystemSettings WHERE SettingKey = 'GPA_HONOR_THRESHOLD')
                 AND NEW.AttendancePercentage >= 90 THEN 'Honors'
            WHEN NEW.CumulativeGPA < (SELECT CAST(SettingValue AS REAL) FROM SystemSettings WHERE SettingKey = 'GPA_PROBATION_THRESHOLD')
                 OR NEW.AttendancePercentage < (SELECT CAST(SettingValue AS REAL) FROM SystemSettings WHERE SettingKey = 'ATTENDANCE_CRITICAL_PERCENT') THEN 'Critical'
            WHEN NEW.CumulativeGPA < (SELECT CAST(SettingValue AS REAL) FROM SystemSettings WHERE SettingKey = 'GPA_WARNING_THRESHOLD')
                 OR NEW.AttendancePercentage < (SELECT CAST(SettingValue AS REAL) FROM SystemSettings WHERE SettingKey = 'ATTENDANCE_WARNING_PERCENT') THEN 'At Risk'
            ELSE 'Normal'
        END,
        LastModifiedDate = CURRENT_TIMESTAMP
    WHERE StudentID = NEW.StudentID;

    INSERT INTO StudentAcademicStatus (StudentID, TermID, Semester, AcademicYear,
                                       AcademicStanding, SemesterGPA, CumulativeGPA,
                                       CreditsAttempted, CreditsEarned, PerformanceFlag)
    SELECT
        NEW.StudentID,
        at.TermID,
        at.Semester,
        at.AcademicYear,
        (SELECT AcademicStanding FROM Student WHERE StudentID = NEW.StudentID),
        NEW.SemesterGPA,
        NEW.CumulativeGPA,
        (SELECT SUM(c.Credits) FROM Enrollment e
         JOIN Section s ON e.SectionID = s.SectionID
         JOIN Course c ON s.CourseID = c.CourseID
         WHERE e.StudentID = NEW.StudentID AND e.Status IN ('Enrolled', 'Completed') AND s.TermID = at.TermID),
        NEW.TotalCreditsEarned,
        (SELECT PerformanceFlag FROM Student WHERE StudentID = NEW.StudentID)
    FROM AcademicTerm at
    WHERE at.IsCurrent = 1
    LIMIT 1;
END;

-- Trigger: Calculate semester GPA and CUMULATIVE GPA on course completion
-- FIX: The previous CumulativeGPA subquery only averaged the grades in the Enrollment
-- table, completely ignoring credits earned BEFORE the current term (e.g. the student's
-- existing 3.2 GPA over 30 credits stored in Student.CumulativeGPA /
-- Student.TotalCreditsEarned).  This caused the trigger to replace the cumulative GPA
-- with just the new course grade points (3.7) instead of the correct weighted average
-- ((3.2*30 + 3.7*3)/33 = 3.2455).
--
-- The correct formula is:
--   new_cumulative = ( old_gpa * old_credits + new_grade_points * course_credits )
--                   / ( old_credits + course_credits )
--
-- We capture the BEFORE-update values via the OLD pseudo-row from the Enrollment update,
-- then read the student's current (pre-update) CumulativeGPA and TotalCreditsEarned
-- directly from the Student table before this trigger modifies them.
CREATE TRIGGER student_semester_gpa_on_completion
AFTER UPDATE OF Status ON Enrollment
WHEN NEW.Status = 'Completed' AND OLD.Status != 'Completed'
BEGIN
    UPDATE Student
    SET SemesterGPA = (
            -- Weighted GPA for courses completed this term only
            SELECT COALESCE(ROUND(SUM(e.GradePoints * c.Credits) / NULLIF(SUM(c.Credits), 0), 2), 0)
            FROM Enrollment e
            JOIN Section s ON e.SectionID = s.SectionID
            JOIN Course c ON s.CourseID = c.CourseID
            WHERE e.StudentID = NEW.StudentID
              AND e.Status = 'Completed'
              AND s.TermID = (SELECT TermID FROM AcademicTerm WHERE IsCurrent = 1 LIMIT 1)
        ),
        TotalCreditsEarned = TotalCreditsEarned + (
            SELECT c.Credits FROM Section s JOIN Course c ON s.CourseID = c.CourseID
            WHERE s.SectionID = NEW.SectionID
        ),
        CumulativeGPA = (
            -- Weighted average: blend previous cumulative GPA with the newly completed course
            SELECT COALESCE(
                ROUND(
                    (
                        (st.CumulativeGPA * st.TotalCreditsEarned)
                        +
                        (NEW_GP.GradePoints * NEW_CR.Credits)
                    )
                    / NULLIF(st.TotalCreditsEarned + NEW_CR.Credits, 0),
                    4
                ),
                0
            )
            FROM Student st
            JOIN (SELECT GradePoints FROM Enrollment WHERE EnrollmentID = NEW.EnrollmentID) AS NEW_GP
            JOIN (SELECT c.Credits FROM Section s JOIN Course c ON s.CourseID = c.CourseID WHERE s.SectionID = NEW.SectionID) AS NEW_CR
            WHERE st.StudentID = NEW.StudentID
        ),
        LastModifiedDate = CURRENT_TIMESTAMP
    WHERE StudentID = NEW.StudentID;
END;

-- Trigger: Check prerequisites with grace period for same-term enrollment
-- FIX: Grace period is now checked against AddDropDeadline (DATE('now') <= AddDropDeadline)
-- instead of julianday('now') - julianday(StartDate) <= PREREQ_GRACE_DAYS.
-- The old formula failed for any term whose StartDate was more than 14 days ago
-- (including the test term that started 2025-09-01), silently blocking the grace path
-- and letting the outer EXISTS fall through without raising — so the INSERT succeeded
-- even when it should have been blocked (Scenario A) AND the grace period never helped
-- (Scenario B).  Using AddDropDeadline is semantically correct: the registrar controls
-- exactly when same-term co-enrollment is allowed.
CREATE TRIGGER enrollment_prerequisite_check
BEFORE INSERT ON Enrollment
WHEN NEW.Status = 'Enrolled'
BEGIN
    SELECT CASE
        WHEN EXISTS (
            SELECT 1 FROM Section s
            JOIN Course c ON s.CourseID = c.CourseID
            JOIN AcademicTerm t ON s.TermID = t.TermID
            WHERE s.SectionID = NEW.SectionID
            AND EXISTS (
                SELECT 1 FROM CoursePrerequisite cp
                WHERE cp.CourseID = c.CourseID
                AND NOT EXISTS (
                    -- Passed the prerequisite in a previous term
                    SELECT 1 FROM Enrollment e_prev
                    JOIN Section s_prev ON e_prev.SectionID = s_prev.SectionID
                    JOIN Course c_prev ON s_prev.CourseID = c_prev.CourseID
                    WHERE e_prev.StudentID = NEW.StudentID
                      AND e_prev.Status = 'Completed'
                      AND e_prev.NumericGrade >= cp.MinGradeRequired
                      AND c_prev.CourseID = cp.PrerequisiteCourseID
                )
                AND NOT EXISTS (
                    -- Grace period: currently enrolled in the prerequisite in the SAME term
                    -- and the add/drop window is still open
                    SELECT 1 FROM Enrollment e_current
                    JOIN Section s_current ON e_current.SectionID = s_current.SectionID
                    JOIN Course c_current ON s_current.CourseID = c_current.CourseID
                    WHERE e_current.StudentID = NEW.StudentID
                      AND e_current.Status = 'Enrolled'
                      AND s_current.TermID = s.TermID
                      AND c_current.CourseID = cp.PrerequisiteCourseID
                      AND DATE('now') <= t.AddDropDeadline
                )
            )
        ) THEN RAISE(ABORT, 'Student has not met the prerequisites for this course')
    END;
END;

-- Trigger: Waitlist — Step 1: insert Waitlist row (runs first, alphabetically before _check)
-- RAISE(ABORT) in a BEFORE trigger rolls back only the triggering INSERT, not sibling
-- BEFORE triggers that already executed.  So we split into two triggers:
--   enrollment_waitlist_add  — writes the Waitlist row (no RAISE)
--   enrollment_waitlist_check — raises ABORT to cancel the Enrollment INSERT
CREATE TRIGGER enrollment_waitlist_add
BEFORE INSERT ON Enrollment
WHEN (SELECT CAST(SettingValue AS TEXT) FROM SystemSettings WHERE SettingKey = 'WAITLIST_ENABLED') = 'true'
AND NEW.Status = 'Enrolled'
AND (SELECT CurrentEnrollment FROM Section WHERE SectionID = NEW.SectionID) >= (SELECT MaxCapacity FROM Section WHERE SectionID = NEW.SectionID)
AND NOT EXISTS (SELECT 1 FROM Waitlist WHERE StudentID = NEW.StudentID AND SectionID = NEW.SectionID)
BEGIN
    INSERT INTO Waitlist (StudentID, SectionID, Position, Status)
    VALUES (
        NEW.StudentID,
        NEW.SectionID,
        COALESCE((SELECT MAX(Position) + 1 FROM Waitlist WHERE SectionID = NEW.SectionID), 1),
        'Waiting'
    );
END;

-- Trigger: Waitlist — Step 2: abort the enrollment (runs after _add)
CREATE TRIGGER enrollment_waitlist_check
BEFORE INSERT ON Enrollment
WHEN (SELECT CAST(SettingValue AS TEXT) FROM SystemSettings WHERE SettingKey = 'WAITLIST_ENABLED') = 'true'
AND NEW.Status = 'Enrolled'
AND (SELECT CurrentEnrollment FROM Section WHERE SectionID = NEW.SectionID) >= (SELECT MaxCapacity FROM Section WHERE SectionID = NEW.SectionID)
BEGIN
    SELECT RAISE(ABORT, 'Section is full. Student has been added to waitlist at position ' ||
        COALESCE(
            (SELECT CAST(Position AS TEXT) FROM Waitlist
             WHERE StudentID = NEW.StudentID AND SectionID = NEW.SectionID),
            '1'
        ));
END;

CREATE TRIGGER course_stats_update
AFTER UPDATE OF NumericGrade, Status ON Enrollment
WHEN NEW.Status = 'Completed' OR OLD.Status != NEW.Status OR OLD.NumericGrade != NEW.NumericGrade

    UPDATE Course

        FROM Enrollment e
        JOIN Section s ON e.SectionID = s.SectionID
        WHERE s.CourseID = Course.CourseID AND e.Status = 'Completed'
    ),
    LastModifiedDate = CURRENT_TIMESTAMP
    WHERE CourseID = (SELECT s.CourseID FROM Section s WHERE s.SectionID = NEW.SectionID);
END;

-- Trigger: Prevent circular prerequisites
CREATE TRIGGER prevent_circular_prerequisites
BEFORE INSERT ON CoursePrerequisite
BEGIN
    WITH RECURSIVE prereq_chain(CourseID, PrerequisiteCourseID, Depth) AS (
        SELECT cp.CourseID, cp.PrerequisiteCourseID, 1
        FROM CoursePrerequisite cp
        WHERE cp.CourseID = NEW.PrerequisiteCourseID
        UNION ALL
        SELECT cp.CourseID, cp.PrerequisiteCourseID, pc.Depth + 1
        FROM CoursePrerequisite cp
        JOIN prereq_chain pc ON cp.CourseID = pc.PrerequisiteCourseID
        WHERE pc.Depth < (SELECT CAST(SettingValue AS INTEGER) FROM SystemSettings WHERE SettingKey = 'MAX_RECURSION_DEPTH')
    )
    SELECT CASE
        WHEN EXISTS (SELECT 1 FROM prereq_chain WHERE PrerequisiteCourseID = NEW.CourseID) THEN
            RAISE(ABORT, 'Circular prerequisite detected for CourseID ' || NEW.CourseID)
    END;
END;

-- Trigger: Dynamic scholarship adjustment based on GPA
CREATE TRIGGER scholarship_dynamic_adjustment
AFTER UPDATE OF CumulativeGPA ON Student
BEGIN
    -- Step 1: Update status based on new GPA
    UPDATE Scholarship
    SET EligibilityStatus = CASE
        WHEN NEW.CumulativeGPA < (SELECT SettingValue FROM SystemSettings WHERE SettingKey = 'SCHOLARSHIP_GPA_THRESHOLD')
             AND EligibilityStatus = 'Active' THEN 'Under Review'
        WHEN NEW.CumulativeGPA >= (SELECT SettingValue FROM SystemSettings WHERE SettingKey = 'SCHOLARSHIP_GPA_THRESHOLD')
             AND EligibilityStatus IN ('Under Review', 'Active') THEN 'Active'
        ELSE EligibilityStatus
    END,
    LastModifiedDate = CURRENT_TIMESTAMP
    WHERE StudentID = NEW.StudentID;

    -- Step 2: Apply amount adjustment for Active scholarships
    UPDATE Scholarship
    SET Amount = CASE
        WHEN NEW.CumulativeGPA >= 4.0 THEN Amount * 1.1
        WHEN NEW.CumulativeGPA >= 3.9 THEN Amount * 1.05
        ELSE Amount
    END,
    LastModifiedDate = CURRENT_TIMESTAMP
    WHERE StudentID = NEW.StudentID
      AND EligibilityStatus = 'Active';

    -- Step 3: Revoke after review period (unchanged)
    UPDATE Scholarship
    SET EligibilityStatus = 'Revoked'
    WHERE StudentID = NEW.StudentID
      AND EligibilityStatus = 'Under Review'
      AND NEW.CumulativeGPA < (SELECT SettingValue FROM SystemSettings WHERE SettingKey = 'SCHOLARSHIP_GPA_THRESHOLD')
      AND LastModifiedDate < DATE('now', '-' || (SELECT SettingValue FROM SystemSettings WHERE SettingKey = 'SCHOLARSHIP_REVIEW_DAYS') || ' days');
END;

-- MATERIALIZED VIEW REFRESH ROUTINE (Manual or Scheduled)
-- =====================================================

-- Refresh all materialized views (call this after bulk changes)
CREATE TEMP TRIGGER IF NOT EXISTS refresh_all_materialized_views
INSTEAD OF INSERT ON dummy_refresh_view
BEGIN
    -- Refresh DepartmentPerformanceSummary
    DELETE FROM DepartmentPerformanceSummary;
    INSERT INTO DepartmentPerformanceSummary (DeptID, DeptName, TotalStudents, TotalCourses, AvgStudentGPA, GraduationRate, TotalBudgetUsed, LastUpdated)
    SELECT
        d.DeptID,
        d.DeptName,
        COUNT(DISTINCT s.StudentID) AS TotalStudents,
        COUNT(DISTINCT c.CourseID) AS TotalCourses,
        COALESCE(AVG(s.CumulativeGPA), 0) AS AvgStudentGPA,
        COALESCE(100.0 * SUM(CASE WHEN s.Status = 'Graduated' THEN 1 ELSE 0 END) / NULLIF(COUNT(s.StudentID), 0), 0) AS GraduationRate,
        COALESCE(SUM(d.BudgetUsed), 0) AS TotalBudgetUsed,
        CURRENT_TIMESTAMP
    FROM Department d
    LEFT JOIN Student s ON d.DeptID = s.DepartmentID AND s.IsDeleted = 0
    LEFT JOIN Course c ON d.DeptID = c.DepartmentID AND c.IsDeleted = 0
    GROUP BY d.DeptID, d.DeptName;

    -- Refresh MaterializedStudentGPAReport
    DELETE FROM MaterializedStudentGPAReport;
    INSERT INTO MaterializedStudentGPAReport
        (StudentID, StudentNumber, StudentName, Country, CurrentSemester,
         CumulativeGPA, SemesterGPA, TotalCreditsEarned, CompletedCourses, HasOutstandingFees, PerformanceFlag, LastRefreshed)
    SELECT
        s.StudentID, s.StudentNumber, s.FirstName || ' ' || s.LastName, s.Country,
        s.CurrentSemester, s.CumulativeGPA, s.SemesterGPA, s.TotalCreditsEarned,
        (SELECT COUNT(*) FROM Enrollment e WHERE e.StudentID = s.StudentID AND e.Status = 'Completed'),
        s.HasOutstandingFees, s.PerformanceFlag, CURRENT_TIMESTAMP
    FROM Student s
    WHERE s.IsDeleted = 0;
END;
